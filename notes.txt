# requete sql qui crée le fichier csv avec les moyennes 
  SELECT
    r.name AS categorie,
    COUNT(DISTINCT r.id) AS nb_runs,
    ROUND(AVG(totaux.energie_cpu_uJ)) AS energie_cpu_moyenne_uJ,
    ROUND(STDDEV(totaux.energie_cpu_uJ)) AS ecart_type_cpu,
    ROUND(AVG(totaux.energie_gpu_uJ)) AS energie_gpu_moyenne_uJ,
    ROUND(AVG(totaux.duree_us)) AS duree_moyenne_us
  FROM runs r JOIN (
    SELECT
      ps.run_id,
      SUM(CASE WHEN ps.metric = 'cores_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_cpu_uJ,
      SUM(CASE WHEN ps.metric = 'gpu_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_gpu_uJ,
      SUM(CASE WHEN ps.metric = 'phase_time_syscall_system' THEN ps.value ELSE 0 END) AS duree_us
    FROM phase_stats ps
    GROUP BY ps.run_id
  ) totaux ON totaux.run_id = r.id
  WHERE r.name IN ('ytb-Ublock', 'ytb-AdGuard', 'ytb-AdBlockPlus', 'ytb-noAddBlock', 'ytb-premium')
  GROUP BY r.name
  ORDER BY energie_cpu_moyenne_uJ DESC 

# ligne de commande complète
psql -h localhost -p 9573 -U postgres -d green-coding -c "\COPY (SELECT r.name AS categorie, COUNT(DISTINCT r.id) AS nb_runs, ROUND(AVG(totaux.energie_cpu_uJ)) AS energie_cpu_moyenne_uJ, ROUND(STDDEV(totaux.energie_cpu_uJ)) AS ecart_type_cpu, ROUND(AVG(totaux.energie_gpu_uJ)) AS energie_gpu_moyenne_uJ, ROUND(AVG(totaux.duree_us)) AS duree_moyenne_us FROM runs r JOIN (SELECT ps.run_id, SUM(CASE WHEN ps.metric = 'cores_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_cpu_uJ, SUM(CASE WHEN ps.metric = 'gpu_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_gpu_uJ, SUM(CASE WHEN ps.metric = 'phase_time_syscall_system' THEN ps.value ELSE 0 END) AS duree_us FROM phase_stats ps GROUP BY ps.run_id) totaux ON totaux.run_id = r.id WHERE r.name IN ('testUblock', 'testAdGuard', 'testAdBlockPlus') GROUP BY r.name ORDER BY energie_cpu_moyenne_uJ DESC) TO '/Users/claracornut/Desktop/moyennes.csv' WITH CSV HEADER"

# requete sql qui crée le fichier csv avec les données de chaque run séparément
# explication : un run = plusieurs "phases" => il faut les sommer pour avoir le résultat
\COPY (
  SELECT
    r.name AS categorie,
    r.id AS run_id,
    r.created_at,
    SUM(CASE WHEN ps.metric = 'cores_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_cpu_uJ,
    SUM(CASE WHEN ps.metric = 'gpu_energy_powermetrics_component' THEN ps.value ELSE 0 END) AS energie_gpu_uJ,
    SUM(CASE WHEN ps.metric = 'phase_time_syscall_system' THEN ps.value ELSE 0 END) AS duree_us
  FROM runs r
  JOIN phase_stats ps ON ps.run_id = r.id
  WHERE r.name IN ('testUblock', 'testAdGuard', 'testAdBlockPlus')
  GROUP BY r.id, r.name, r.created_at
  ORDER BY r.name, r.created_at
) TO '/tmp/runs_detail.csv' WITH CSV HEADER;
